DROP TABLE EMPREGISTER;
DROP TABLE SALARY;
DROP TABLE SCHEDULE;

CREATE TABLE EMPREGISTER ( -- 직원등록 테이블
        EMPREGISTER_NAME VARCHAR (50) NOT NULL, -- 이름
        EMPREGISTER_PHOTO VARCHAR (300), -- 사원 사진 
        EMPREGISTER_EMPNUM VARCHAR(20),  -- 사번(PK)
        EMPREGISTER_AGE INT, -- 나이
        EMPREGISTER_TEL VARCHAR (50), -- 전화번호 
        EMPREGISTER_JUMIN VARCHAR (50), -- 주민1 
        EMPREGISTER_JUMIN2 VARCHAR (50),  -- 주민2
        EMPREGISTER_GENDER VARCHAR (50), -- 성별
        EMPREGISTER_ADDR VARCHAR (100), -- 거주지
        EMPREGISTER_ADDR2 VARCHAR (100), -- 거주지
        EMPREGISTER_ADDR3 VARCHAR (100), -- 거주지
        EMPREGISTER_ADDR4 VARCHAR (100), -- 거주지
        EMPREGISTER_GRADE VARCHAR (50),  -- 직급 
        EMPREGISTER_BANKNUM VARCHAR (100), -- 통장번호 
        EMPREGISTER_BANKNAME VARCHAR (50), -- 은행명 
        EMPREGISTER_ACCOUNTHOLDER VARCHAR(50), -- 예금주  
        EMPREGISTER_ENTRYDAY TIMESTAMP DEFAULT NOW(), -- 입사날짜 
        EMPREGISTER_PAYCLASSFIY VARCHAR(20), -- 급여구분
        EMPREGISTER_PAYTIME VARCHAR(20), -- 시급        
        EMPREGISTER_PAYDAY VARCHAR(20), -- 일급
        EMPREGISTER_WORKPLAN INT, -- 근무예정시간
        EMPREGISTER_LEAVECOMPANY VARCHAR(20), -- 퇴사여부 
        EMPREGISTER_ENABLED INT DEFAULT 1, -- 사용여부
        EMPREGISTER_AUTHORITY VARCHAR(20) DEFAULT 'ROLE_MEMBER'-- 권한
);

CREATE TABLE SALARY ( -- 근무 및 급여내역, 저장하는테이블
        SALARY_NO VARCHAR(20), -- 번호(PK)
        SALARY_EMPREGISTER_EMPNUM VARCHAR(20), -- 사번(FK)
        SALARY_WORKINGTIME INT, -- 일한시간
        SALARY_WORKINGDATE INT, -- 근무일수
        SALARY_HOLLYDAYSTART TIMESTAMP, -- 휴가시작
        SALARY_HOLLYDAYEND TIMESTAMP, -- 휴가종료
        SALARY_PAY INT
);

CREATE TABLE SCHEDULE ( -- 스케쥴
        SCHEDULE_NO VARCHAR(20), -- 번호(PK)
        SCHEDULE_EMPREGISTER_EMPNUM VARCHAR(20),   -- 사번(FK)
        SCHEDULE_WORKINGSTART TIMESTAMP, -- 출근시간
        SCHEDULE_WORKINGEND TIMESTAMP, -- 퇴근시간
        SCHEDULE_WORKINGTIME INT, -- 일한시간 (매달 초기화)
        SCHEDULE_WORKINGDATE INT -- 당월근무일수 (퇴근 카운트, 매달 초기화)
);



ALTER TABLE EMPREGISTER ADD PRIMARY KEY(EMPREGISTER_EMPNUM);
ALTER TABLE SALARY ADD PRIMARY KEY(SALARY_NO);
ALTER TABLE SCHEDULE ADD PRIMARY KEY(SCHEDULE_NO);

ALTER TABLE SALARY ADD CONSTRAINT FK_SALARY_EMPREGISTER_EMPNUM FOREIGN KEY(SALARY_EMPREGISTER_EMPNUM) REFERENCES EMPREGISTER(EMPREGISTER_EMPNUM);
ALTER TABLE SCHEDULE ADD CONSTRAINT FK_SCHEDULE_EMPREGISTER_EMPNUM FOREIGN KEY(SCHEDULE_EMPREGISTER_EMPNUM) REFERENCES EMPREGISTER(EMPREGISTER_EMPNUM);
